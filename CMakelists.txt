cmake_minimum_required(VERSION 3.14)
project(json_parser LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_library(json STATIC src/json.cpp src/json.h)

add_executable(main src/main.cpp)
target_link_libraries(main PRIVATE json)

option(BUILD_TESTS "Build tests" ON)
option(LOCAL_BUILD "Enable if building without internet (uses extern/ dependencies)" ON)

if(BUILD_TESTS)
    enable_testing()
    # Собираем тесты
    if(LOCAL_BUILD)
        add_subdirectory(extern/googletest)
    else()
        FetchContent_Declare(
            googletest
            URL https://github.com/google/googletest/archive/refs/tags/v1.14.0.zip
        )
        # Для пользователей: не устанавливаем gtest в систему
        set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
        FetchContent_MakeAvailable(googletest)
    endif()

    add_subdirectory(tests)
    
endif()
